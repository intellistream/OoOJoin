96  vector<tsType> genZipfInt(size_t len, tsType maxV, double fac) {
97  vector<tsType> ret(len);
98  vector<tsType> alphabet = genIncrementalAlphabet<tsType>(maxV);
99  std::mt19937_64 gen;
100  if (!hasSeed) {
101  gen = std::mt19937_64(rd()); // 以 rd() 播种的标准 mersenne_twister_engine
102  } else {
103  gen = std::mt19937_64(seed);
104  }
105 
106  std::uniform_real_distribution<> dis(0, 1);
107  vector<double> lut = genZipfLut<double>(maxV, fac);
108  for (size_t i = 0; i < len; i++) {
109  /* take random number */
110  double r = dis(gen);
111  /* binary search in lookup table to determine item */
112  size_t left = 0;
113  size_t right = maxV - 1;
114  size_t m; /* middle between left and right */
115  size_t pos; /* position to take */
116 
117  if (lut[0] >= r)
118  pos = 0;
119  else {
120  while (right - left > 1) {
121  m = (left + right) / 2;
122 
123  if (lut[m] < r)
124  left = m;
125  else
126  right = m;
127  }
128 
129  pos = right;
130  }
131  ret[i] = alphabet[pos];
132  }
133  return ret;
134  }

172  template<class dType=double>
173  vector<dType> genZipfLut(size_t len, dType fac) {
174  dType scaling_factor;
175  dType sum;
176  vector<dType> lut(len);
177  /*
178  * Compute scaling factor such that
179  *
180  * sum (lut[i], i=1..alphabet_size) = 1.0
181  *
182  */
183  scaling_factor = 0.0;
184  for (size_t i = 1; i <= len; i++) { scaling_factor += 1.0 / pow(i, fac); }
185  /*
186  * Generate the lookup table
187  */
188  sum = 0.0;
189  for (size_t i = 1; i <= len; i++) {
190  sum += 1.0 / std::pow(i, fac);
191  lut[i - 1] = sum / scaling_factor;
192  }
193  return lut;
194  }
